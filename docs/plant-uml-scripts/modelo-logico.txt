@startuml
entity "Usuario" {
  + id : String <<PK>>
  nome : String
  email : String
  senha : String
  funcao : FuncaoDeUsuario
  criado : Date
}

entity "Post" {
  + id : String <<PK>>
  titulo : String
  corpo : String
  criadorId : String <<FK>>
  criado : Date
  ultimaAtualizacao : Date
  imagemBase64 : String
}

entity "Comentario" {
  + id : String <<PK>>
  usuarioId : String <<FK>>
  idComentarioPai : String <<FK>>
  tipoEntidadePai : String
  idPots : String <<FK>>
  dataComentario : Date
}

entity "Oportunidade" {
  + id : String <<PK>>
  professorId : String <<FK>>
  nome : String
  descricao : String
  quantidadeDeVagas : int
  vagasPreenchidas : int
  criado : Date
  idCategoria : String <<FK>>
}

entity "Categoria" {
  + id : String <<PK>>
  nome : String
}

entity "Imagem" {
  + id : String <<PK>>
  url : String
}
